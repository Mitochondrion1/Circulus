1    package com.circulus.playtime;
2    
3    import android.app.Service;
4    import android.content.BroadcastReceiver;
5    import android.content.Context;
6    import android.content.Intent;
7    import android.content.IntentFilter;
8    import android.media.MediaPlayer;
9    import android.net.ConnectivityManager;
10   import android.os.IBinder;
11   import android.widget.Toast;
12   
13   import com.circulus.main.R;
14   import com.circulus.utility.Store;
15   
16   /**
17    * A service that plays music during gameplay.
18    */
19   public class MusicService extends Service {
20       /** The media player object. */
21       private MediaPlayer mPlayer;
22   
23       /** The broadcast receiver. */
24       HeadsetIntentReceiver receiver;
25   
26       /** The headset volume value. */
27       private int headsetVolume;
28       /** The speaker volume value. */
29       private int speakerVolume;
30   
31       @Override
32       public void onCreate() {
33           super.onCreate();
34   
35           // Create the broadcast receiver, Register the media player and read the volumes from shared preferences
36           receiver = new HeadsetIntentReceiver();
37           mPlayer = MediaPlayer.create(this, R.raw.project_theme);
38           headsetVolume = Store.readInt(getApplicationContext(), R.string.headset_volume_key, 20);
39           speakerVolume = Store.readInt(getApplicationContext(), R.string.speaker_volume_key, 100);
40   
41           if (null != mPlayer) {
42               mPlayer.setLooping(true);
43   
44               IntentFilter receiverFilter = new IntentFilter(ConnectivityManager.CONNECTIVITY_ACTION);
45               receiverFilter.addAction(Intent.ACTION_HEADSET_PLUG);
46               registerReceiver(receiver, receiverFilter);
47           }
48       }
49   
50       /**
51        * Set the volume to be the headset volume.
52        */
53       public void setVolumeHeadset(){
54           mPlayer.setVolume(headsetVolume / 100f, headsetVolume / 100f);
55       }
56   
57       /**
58        * Set the volume to be the speaker volume.
59        */
60       public void setVolumeSpeaker(){
61           mPlayer.setVolume(speakerVolume / 100f, speakerVolume / 100f);
62       }
63   
64       @Override
65       public int onStartCommand(Intent intent, int flags, int startid) {
66   
67           if (null != mPlayer) {
68               if (mPlayer.isPlaying()) {
69                   // Rewind to beginning of song
70                   mPlayer.seekTo(0);
71               }
72               else {
73                   // Start playing song
74                   mPlayer.start();
75               }
76           }
77   
78           // Don't automatically restart this Service if it is killed
79           return START_NOT_STICKY;
80       }
81   
82       @Override
83       public IBinder onBind(Intent intent) {
84           return null;
85       }
86   
87       @Override
88       public void onDestroy() {
89           // Stop playing music
90           super.onDestroy();
91           mPlayer.stop();
92           unregisterReceiver(receiver);
93           stopForeground(true);
94       }
95   
96       /**
97        * Update the state for the volume (for when a headset is plugged/unplugged).
98        * <p>
99        * @param state The state as it is received from the broadcast receiver.
100       */
101      private void updateState(String state) {
102          Toast.makeText(this, state, Toast.LENGTH_LONG).show();
103          if (state.equals("Plugged")) {
104              setVolumeHeadset();
105          }
106          else {
107              setVolumeSpeaker();
108          }
109      }
110  
111      /**
112       * The broadcast receiver that receives a headset plug/unplug broadcast.
113       */
114      public class HeadsetIntentReceiver extends BroadcastReceiver {
115          @Override
116          public void onReceive(Context context, Intent intent) {
117              if (intent.getAction().equals(Intent.ACTION_HEADSET_PLUG)) {
118                  int state = intent.getIntExtra("state", -1);
119                  switch (state) {
120                      case (0):
121                          updateState("Unplugged");
122                          break;
123                      case (1):
124                          updateState("Plugged");
125                          break;
126                      default:
127                          updateState("Error");
128                  }
129              }
130          }
131      }
132  }
