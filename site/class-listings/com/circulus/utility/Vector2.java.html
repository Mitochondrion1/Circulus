1    package com.circulus.utility;

3    /**
4     * Defines a 2-dimensional vector object, with float coordinates.
5     */
6    public class Vector2 {
7        // Load native-lib.cpp
8        static {
9            System.loadLibrary("native-lib");
10       }
11   
12       /** The x coordinate. */
13       private float x;
14       /** The y coordinate. */
15       private float y;
16   
17       /**
18        * Constructs a Vector2 object.
19        * <p>
20        * @param x The x coordinate.
21        * @param y The y coordinate.
22        */
23       public Vector2(float x, float y) {
24           this.x = x;
25           this.y = y;
26       }
27   
28       /**
29        * Get the x coordinate.
30        * <p>
31        * @return The x coordinate of the vector.
32        */
33       public float getX() {
34           return x;
35       }
36   
37       /**
38        * Set a new x coordinate for the vector.
39        * <p>
40        * @param x The new value.
41        */
42       public void setX(float x) {
43           this.x = x;
44       }
45   
46       /**
47        * Get the y coordinate.
48        * <p>
49        * @return The y coordinate of the vector.
50        */
51       public float getY() {
52           return y;
53       }
54   
55       /**
56        * Set a new y coordinate for the vector.
57        * <p>
58        * @param y The new value.
59        */
60       public void setY(float y) {
61           this.y = y;
62       }
63   
64       /**
65        * Calculates the length of the vector.
66        * <p>
67        * @return The length of the vector.
68        */
69       public float getLength() {
70           return (float)Math.sqrt(x * x + y * y);
71       }
72   
73       /**
74        * Set the length of the vector to a certain length.
75        * <p>
76        * @param length The desired length.
77        */
78       public void setLength(float length) {
79           // New length calculation uses the algorithm from native-lib
80           float multiplier = length * quickInvSqrt(this.x * this.x + this.y * this.y);
81           this.x *= multiplier;
82           this.y *= multiplier;
83       }
84   
85       /**
86        * Calculates the distance between two given points.
87        * <p>
88        * @param v1    One point.
89        * @param v2    Another point.
90        * @return      The distance between the points.
91        */
92       public static float distance(Vector2 v1, Vector2 v2) {
93           return (float)Math.sqrt((v1.x - v2.x) * (v1.x - v2.x) + (v1.y - v2.y) * (v1.y - v2.y));
94       }
95   
96       /**
97        * Subtracts 2 vectors.
98        * <p>
99        * @param v1    One vector.
100       * @param v2    Another Vector.
101       * @return      The subtraction of the first vector from the second one.
102       */
103      public static Vector2 sub(Vector2 v1, Vector2 v2) {
104          return new Vector2(v2.x - v1.x, v2.y - v1.y);
105      }
106  
107      /**
108       * Get a string representing the vector.
109       * <p>
110       * @return The vector string in a (x,y) format.
111       */
112      @Override
113      public String toString() {
114          return "(" + x + ", " + y + ")";
115      }
116  
117      /**
118       * C++ fast inverse square root algorithm implementation.
119       * <p><
120       * @param lengthSquared The length of the vector squared.
121       * @return              The inverse of the length.
122       */
123      public native float quickInvSqrt(float lengthSquared);
124  }
