1    package com.circulus.gameobject;
2    
3    import android.graphics.Canvas;
4    import android.graphics.Color;
5    import android.graphics.Paint;
6    
7    import com.circulus.playtime.MainView;
8    import com.circulus.playtime.Manager;
9    import com.circulus.utility.Vector2;
10   
11   /**
12    * Defines the Exploder enemy type. Explodes when close to the player. Damages the player are in the explosion AoE.
13    */
14   public class Exploder extends Enemy {
15       /** True if the explosion has been triggered already, otherwise false. */
16       private boolean exploding;
17       /** The amount of ticks left until the explosion. */
18       private int explosionCountdown;
19       /** The radius of the explosion AoE. */
20       private float aoeRadius;
21   
22       /** The paint of the AoE. */
23       private Paint aoePaint;
24   
25       /**
26        * Constructs an Exploder enemy.
27        * <p>
28        * @param position  The position of the enemy.
29        * @param manager   The manager associated with the enemy.
30        * @param view      The main game view.
31        */
32       public Exploder(Vector2 position, Manager manager, MainView view) {
33           super(position, manager, view);
34   
35           this.speed = 0.5f;
36           this.size = 0.3f;
37           this.exploding = false;
38           this.explosionCountdown = 0;
39           this.aoeRadius = 0.75f * (this.size + manager.getPlayer().getSize());
40           this.paint.setColor(Color.CYAN);
41   
42           this.aoePaint = new Paint();
43           this.aoePaint.setColor(Color.CYAN);
44           this.aoePaint.setAlpha(0);
45   
46           this.assignBasicValues(40f, 20f, 2);
47   
48           this.damage = baseDamage * (0.8f + 0.2f * this.manager.getLevel());
49           this.health = baseHealth * (0.9f + 0.1f * this.manager.getLevel() * this.manager.getLevel());
50           this.maxHealth = this.health;
51       }
52   
53       /**
54        * Draws the enemy.
55        * <p>
56        * @param canvas The canvas used for drawing.
57        */
58       @Override
59       public void draw(Canvas canvas) {
60           canvas.drawCircle(this.findPixelPosition().getX(), this.findPixelPosition().getY(),
61                   aoeRadius * view.getPixelsPerUnit(), aoePaint);
62           super.draw(canvas);
63       }
64   
65       /**
66        * {@inheritDoc}
67        */
68       @Override
69       public boolean isAlive() {
70           return this.alive;
71       }
72   
73       /**
74        * Defines the behavior of the Exploder.
75        */
76       @Override
77       protected void behave() {
78           // Define the movement of the Exploder: homing the player from any distance
79           if (Vector2.distance(position, manager.getPlayer().getPosition()) != 0) {
80               move();
81           }
82   
83           // Define what happens after the explosion begins
84           if (this.exploding) {
85               this.explosionCountdown--;
86               if (explosionCountdown > 0) {
87                   this.paint.setARGB(255, (int)((1 - explosionCountdown / 50f) * 255), 255, 255);
88               }
89               else if (this.explosionCountdown == 0) {
90                   if (Vector2.distance(position, manager.getPlayer().getPosition()) <
91                           0.75f * (this.size + manager.getPlayer().getSize()) + 0.5f * manager.getPlayer().getSize()) {
92                       manager.getPlayer().damage(this.damage);
93                   }
94                   this.alive = false;
95               }
96           }
97   
98           // Start an explosion
99           else if (Vector2.distance(position, manager.getPlayer().getPosition()) <
100                      0.75f * (this.size + manager.getPlayer().getSize())) {
101              this.exploding = true;
102              this.explosionCountdown = 50;
103              this.aoePaint.setAlpha(100);
104          }
105      }
106  }
