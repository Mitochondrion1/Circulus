1    package com.circulus.activity;

3    import androidx.appcompat.app.AppCompatActivity;
4    
5    import android.content.pm.ActivityInfo;
6    import android.os.Bundle;

8    import android.view.MenuInflater;
9    import android.view.MenuItem;
10   import android.widget.SeekBar;
11   import android.widget.Switch;
12   
13   import com.circulus.main.R;
14   import com.circulus.utility.Store;
15   
16   /**
17    * The activity that allows to change settings.
18    */
19   public class SettingsActivity extends AppCompatActivity {
20       /** Headset volume slider. */
21       private SeekBar headsetVolume;
22       /** Speaker volume slider. */
23       private SeekBar speakerVolume;
24       /** Accelerometer mode switch. */
25       private Switch accSwitch;
26       /** Direction display switch. */
27       private Switch dirSwitch;
28       /** Accelerometer sensitivity slider. */
29       private SeekBar accSensitivity;
30   
31       /** A value representing the current layout: 0 - volume, 1 - accessibility. */
32       private int currentView;
33   
34       @Override
35       protected void onCreate(Bundle savedInstanceState) {
36           super.onCreate(savedInstanceState);
37           setRequestedOrientation(ActivityInfo.SCREEN_ORIENTATION_LANDSCAPE);
38           setContentView(R.layout.settings_volume);
39           getSupportActionBar().setTitle(R.string.button_settings);
40           currentView = 0;
41           headsetVolume = findViewById(R.id.headsetVol);
42           speakerVolume = findViewById(R.id.speakerVol);
43           headsetVolume.setMax(100);
44           speakerVolume.setMax(100);
45           headsetVolume.setProgress(Store.readInt(getApplicationContext(), R.string.headset_volume_key, 20));
46           speakerVolume.setProgress(Store.readInt(getApplicationContext(), R.string.speaker_volume_key, 100));
47       }
48   
49       /**
50        * Inflates the options menu.
51        * <p>
52        * @param menu  The menu to inflate.
53        * @return      true (has no meaning).
54        */
55       @Override
56       public boolean onCreateOptionsMenu(Menu menu) {
57           // Inflate the menu
58           MenuInflater inflater = getMenuInflater();
59           inflater.inflate(R.menu.settings_menu, menu);
60           return true;
61       }
62   
63       /**
64        * Defines what happens when an options menu item is selected.
65        * <p>
66        * @param item  The menu item that is selected.
67        * @return      Returned value has no meaning.
68        */
69       @Override
70       public boolean onOptionsItemSelected(MenuItem item) {
71           // Handle menu item selection
72           switch (item.getItemId()) {
73               case R.id.volume_option:
74                   if (currentView != 0) {
75                       saveSettings();
76                       setContentView(R.layout.settings_volume);
77                       currentView = 0;
78                       headsetVolume = findViewById(R.id.headsetVol);
79                       speakerVolume = findViewById(R.id.speakerVol);
80                       headsetVolume.setMax(100);
81                       speakerVolume.setMax(100);
82                       headsetVolume.setProgress(Store.readInt(getApplicationContext(), R.string.headset_volume_key, 20));
83                       speakerVolume.setProgress(Store.readInt(getApplicationContext(), R.string.speaker_volume_key, 100));
84                   }
85                   return true;
86               case R.id.accessibility_option:
87                   if (currentView != 1) {
88                       saveSettings();
89                       setContentView(R.layout.settings_accessibility);
90                       currentView = 1;
91                       accSwitch = findViewById(R.id.accelerometer_switch);
92                       dirSwitch = findViewById(R.id.direction_switch);
93                       accSensitivity = findViewById(R.id.accSensitivity);
94                       accSwitch.setChecked(Store.readBool(getApplicationContext(), R.string.accelerometer_key, true));
95                       dirSwitch.setChecked(Store.readBool(getApplicationContext(), R.string.direction_display_key, false));
96                       accSensitivity.setMax(3);
97                       accSensitivity.setProgress(Store.readInt(getApplicationContext(), R.string.sensitivity_key, 2) - 1);
98                   }
99                   return true;
100              case R.id.to_menu_option:
101                  finish();
102              default:
103                  return super.onOptionsItemSelected(item);
104          }
105      }
106  
107      // Save settings when the activity is stopped, destroyed or paused
108      @Override
109      protected void onStop() {
110          super.onStop();
111          saveSettings();
112      }
113  
114      @Override
115      protected void onDestroy() {
116          super.onDestroy();
117          saveSettings();
118      }
119  
120      @Override
121      protected void onPause() {
122          super.onPause();
123          saveSettings();
124      }
125  
126  
127      /**
128       * Saves the settings.
129       */
130      private void saveSettings() {
131          if (currentView == 0) {
132              Store.saveInt(getApplicationContext(), R.string.headset_volume_key, headsetVolume.getProgress());
133              Store.saveInt(getApplicationContext(), R.string.speaker_volume_key, speakerVolume.getProgress());
134          }
135          if (currentView == 1) {
136              Store.saveBool(getApplicationContext(), R.string.accelerometer_key, accSwitch.isChecked());
137              Store.saveBool(getApplicationContext(), R.string.direction_display_key, dirSwitch.isChecked());
138              Store.saveInt(getApplicationContext(), R.string.sensitivity_key, accSensitivity.getProgress() + 1);
139          }
140      }
